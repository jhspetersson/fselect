Example Queries

select name, size from /sandbox
select name, size where ext = rs
select path, size where size > 10k order by size desc
select name where name like '%.md'
select name, is_dir where is_dir = true
select count(*) where is_file = true
select name, modified order by modified desc limit 5
select name, size where name = '*.rs' or name = '*.py'
select name from ... as parent where is_dir and exists (select ... where child.dir = parent.path and is_source)
select name from .../projects where name in (select name from .../home)
select name, (select count(*) ... where inner.dir = outer.path) where is_dir
select name, ext from .../projects where ext in (select ext from .../documents)
